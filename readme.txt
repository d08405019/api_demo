git init #建立數據庫( 必須再預計建立資料夾內，才執行指令 ) 
git config --global user.name "<Your Name>"#連結到哪一個使用者
git config --global user.email "<your@gmail.com>"#設定電子郵件
git add . or git add index.html #將異動的資料 加入索引 
git commit –m '修改紀錄說明' #提交更新
git remote add origin '儲存位置'#儲存位置設定
git push -u origin master #上傳檔案
git clone '網址'#下載檔案
